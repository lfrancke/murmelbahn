# Frontend SSR Rebuild Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Replace the Svelte SPA with server-rendered, crawlable HTML pages (askama on axum 0.8), keeping the existing JSON/CSV API byte-for-byte intact.

**Architecture:** Phase 1 migrates the web crate from axum 0.6 to 0.8 with the current API unchanged and verified. Phase 2 adds askama page handlers (`/`, `/course/{code}`, `/sets`, `/build`, `/about`), serves hand-written CSS + one vanilla-JS enhancement file as static assets, deletes the `frontend/` Svelte project, and makes the Dockerfile Rust-only. English-only; no browse/search, no 3D viewer (each is a later spec).

**Tech Stack:** Rust (edition 2024), axum 0.8, askama 0.12, tower-http 0.6, sqlx 0.8, Postgres, `murmelbahn-lib`.

---

## Conventions for the whole plan

- Run every `cargo` and `git` command with `TMPDIR=$PWD/.build-tmp` (the sandbox `/tmp` is read-denied). Create `.build-tmp/` once if missing: `mkdir -p .build-tmp`.
- Do NOT use shell heredocs. Author files with the editor, not `cat <<EOF`.
- No em dashes anywhere (code, comments, docs, commit messages). Use commas, colons, parentheses, or periods.
- Commit messages: plain and factual, imperative subject, no AI boilerplate, no "Generated by" trailer.
- Run `TMPDIR=$PWD/.build-tmp cargo fmt` and `TMPDIR=$PWD/.build-tmp cargo clippy --workspace` before every commit; land formatted and clippy-clean.
- The Android-app decompilation must never be referenced in any tracked file, comment, or commit message.
- Branch: do this on a feature branch off `main` (e.g. `frontend-ssr`). Phase 1 and Phase 2 are separate PRs.

```rust
use askama::Template;
use axum::http::StatusCode;
use axum::response::{Html, IntoResponse, Response};

/// One error type for the HTML page handlers. Renders a friendly page instead
/// of leaking internals or panicking.
#[derive(Debug)]
pub enum AppError {
    NotFound(String),
    Internal(String),
}

#[derive(Template)]
#[template(path = "error.html")]
struct ErrorTemplate {
    status: u16,
    message: String,
}

impl IntoResponse for AppError {
    fn into_response(self) -> Response {
        let (code, message) = match self {
            AppError::NotFound(m) => (StatusCode::NOT_FOUND, m),
            AppError::Internal(m) => (StatusCode::INTERNAL_SERVER_ERROR, m),
        };
        let body = ErrorTemplate {
            status: code.as_u16(),
            message,
        }
        .render()
        .unwrap_or_else(|_| "error".to_string());
        (code, Html(body)).into_response()
    }
}
```

```rust
use murmelbahn_lib::app::BillOfMaterials;

/// A labelled group of pieces for the BOM table (Tiles, Rails, Walls, Layers,
/// Connectors).
pub struct BomSection {
    pub title: &'static str,
    pub rows: Vec<BomRow>,
}

pub struct BomRow {
    pub label: String,
    pub count: i32,
}

fn rows_from<K: std::fmt::Debug>(map: &std::collections::HashMap<K, i32>) -> Vec<BomRow> {
    let mut rows: Vec<BomRow> = map
        .iter()
        .filter(|(_, &c)| c > 0)
        .map(|(k, &c)| BomRow {
            label: format!("{k:?}"),
            count: c,
        })
        .collect();
    // Sort by count descending, then label ascending for a stable order.
    rows.sort_by(|a, b| b.count.cmp(&a.count).then_with(|| a.label.cmp(&b.label)));
    rows
}

/// Builds the display sections for a bill of materials. Only non-empty groups
/// are returned.
pub fn sections(bom: &BillOfMaterials) -> Vec<BomSection> {
    let mut sections = Vec::new();

    for (title, rows) in [
        ("Tiles", rows_from(&bom.tiles)),
        ("Rails", rows_from(&bom.rails)),
        ("Walls", rows_from(&bom.walls)),
        ("Layers", rows_from(&bom.layers)),
    ] {
        if !rows.is_empty() {
            sections.push(BomSection { title, rows });
        }
    }

    if bom.connectors > 0 {
        sections.push(BomSection {
            title: "Connectors",
            rows: vec![BomRow {
                label: "Connector".to_string(),
                count: bom.connectors,
            }],
        });
    }

    sections
}
```

```rust
use askama::Template;
use axum::extract::{Path, State};
use axum::response::Html;
use chrono::DateTime;
use std::sync::Arc;

use murmelbahn_lib::app::BillOfMaterials;
use murmelbahn_lib::app::course::SavedCourse;
use murmelbahn_lib::common::CourseCode;

use crate::AppState;
use crate::bom_view::{self, BomSection};
use crate::error::AppError;

#[derive(Template)]
#[template(path = "course.html")]
struct CourseTemplate {
    code: String,
    title: String,
    created: String,
    version: String,
    marbles: i32,
    sections: Vec<BomSection>,
}

pub async fn course(
    Path(code): Path<String>,
    State(state): State<Arc<AppState>>,
) -> Result<Html<String>, AppError> {
    let course_code = CourseCode::new(code.to_uppercase());

    let bytes = state
        .course_repo
        .get_course_bytes(&course_code)
        .await
        .map_err(|e| AppError::Internal(format!("Could not load course: {e}")))?;

    let Some(bytes) = bytes else {
        return Err(AppError::NotFound(format!(
            "Course [{course_code}] could not be found."
        )));
    };

    let saved = SavedCourse::from_bytes(&bytes)
        .map_err(|e| AppError::Internal(format!("Could not parse course: {e}")))?;

    let meta = saved.course.meta_data();
    let version = format!("{:?}", saved.header.version);
    let created = DateTime::from_timestamp_millis(meta.creation_timestamp as i64)
        .map(|dt| dt.format("%Y-%m-%d").to_string())
        .unwrap_or_else(|| "unknown".to_string());

    let bom = BillOfMaterials::from(saved.course);
    let marbles = bom.marbles();
    let sections = bom_view::sections(&bom);

    Ok(Html(
        CourseTemplate {
            code: course_code.to_string(),
            title: meta.title,
            created,
            version,
            marbles,
            sections,
        }
        .render()
        .unwrap_or_default(),
    ))
}
```

```rust
use askama::Template;
use axum::extract::State;
use axum::response::Html;
use std::sync::Arc;

use crate::AppState;

#[derive(Template)]
#[template(path = "sets.html")]
struct SetsTemplate {
    sets: Vec<SetView>,
}

struct SetView {
    id: String,
    name: String,
    pieces: Vec<PieceRow>,
}

struct PieceRow {
    label: String,
    count: i32,
}

pub async fn sets(State(state): State<Arc<AppState>>) -> Html<String> {
    let mut sets: Vec<SetView> = state
        .sets_repo
        .sets
        .values()
        .map(|s| {
            let name = s
                .names
                .iter()
                .find(|n| n.language_code == "en")
                .or_else(|| s.names.first())
                .map(|n| n.name.clone())
                .unwrap_or_else(|| s.id.clone());

            let mut pieces: Vec<PieceRow> = s
                .content
                .iter()
                .map(|(element, &count)| PieceRow {
                    label: format!("{element:?}"),
                    count,
                })
                .collect();
            pieces.sort_by(|a, b| b.count.cmp(&a.count).then_with(|| a.label.cmp(&b.label)));

            SetView {
                id: s.id.clone(),
                name,
                pieces,
            }
        })
        .collect();

    sets.sort_by(|a, b| a.name.cmp(&b.name));

    Html(SetsTemplate { sets }.render().unwrap_or_default())
}
```

```rust
use askama::Template;
use axum::extract::State;
use axum::response::Html;
use axum::Form;
use std::collections::HashMap;
use std::sync::Arc;

use murmelbahn_lib::physical::Inventory;

use crate::AppState;

struct SetField {
    id: String,
    name: String,
}

struct BuiltCourse {
    code: String,
    title: String,
}

#[derive(Template)]
#[template(path = "build.html")]
struct BuildTemplate {
    sets: Vec<SetField>,
    result: Option<Vec<BuiltCourse>>,
}

/// Turns the posted form (set id -> count string) into an `Inventory`,
/// dropping blank, zero, and non-numeric entries.
fn inventory_from_form(form: HashMap<String, String>) -> Inventory {
    let sets = form
        .into_iter()
        .filter_map(|(id, raw)| match raw.trim().parse::<i32>() {
            Ok(n) if n > 0 => Some((id, n)),
            _ => None,
        })
        .collect();

    Inventory {
        sets,
        extra_elements: HashMap::new(),
    }
}

fn set_fields(state: &AppState) -> Vec<SetField> {
    let mut fields: Vec<SetField> = state
        .sets_repo
        .sets
        .values()
        .map(|s| {
            let name = s
                .names
                .iter()
                .find(|n| n.language_code == "en")
                .or_else(|| s.names.first())
                .map(|n| n.name.clone())
                .unwrap_or_else(|| s.id.clone());
            SetField {
                id: s.id.clone(),
                name,
            }
        })
        .collect();
    fields.sort_by(|a, b| a.name.cmp(&b.name));
    fields
}

pub async fn build_form(State(state): State<Arc<AppState>>) -> Html<String> {
    let sets = set_fields(&state);
    Html(
        BuildTemplate { sets, result: None }
            .render()
            .unwrap_or_default(),
    )
}

pub async fn build_submit(
    State(state): State<Arc<AppState>>,
    Form(form): Form<HashMap<String, String>>,
) -> Html<String> {
    let inventory = inventory_from_form(form);

    let built = state
        .course_repo
        .process_all(&state.sets_repo, inventory)
        .await
        .map(|courses| {
            courses
                .into_iter()
                .map(|c| BuiltCourse {
                    code: c.course_code,
                    title: c.title,
                })
                .collect::<Vec<_>>()
        })
        .unwrap_or_default();

    let sets = set_fields(&state);
    Html(
        BuildTemplate {
            sets,
            result: Some(built),
        }
        .render()
        .unwrap_or_default(),
    )
}
```
